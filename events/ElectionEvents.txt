# 絶対民主主義　支持（ジオン）
country_event = {
	id = election.1
	title = election.1.t
	desc = election.1.d
	picture = GFX_election

	is_triggered_only = yes

	trigger = {
    original_tag = ZEO
    has_government = zeonism
    AND = {
      democratic > 0.3
			democratic < 0.51
      spacesupremacy < 0.15
      earthsupremacy < 0.15
    }
	}

	option = {
		ai_chance = {
			factor = 0.6
		}
		name = election.1.a
	}

	option = {
		name = election.1.b
		ai_chance = {
			factor = 0.4
		}
    add_popularity = {
      ideology = democratic
      popularity = -0.05
    }
    add_stability = -0.05
	}
}
