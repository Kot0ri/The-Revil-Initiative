enemy_has_equal_or_better_radar_tech_trigger = {
	custom_trigger_tooltip = {
		tooltip = enemy_has_equal_or_better_radar_tech_trigger_tt
		if = {
			limit = {
				ROOT = {
					has_tech = mpiws1
					NOT = {
						has_tech = mpiws2
					}
				}
			}
			FROM = {
				OR = {
					has_tech = mpiws1
					has_tech = mpiws2
				}
			}
		}
		else_if = {
			limit = {
				ROOT = {
					has_tech = mpiws2
					NOT = {
						has_tech = mpiws3
					}
				}
			}
			FROM = {
				OR = {
					has_tech = mpiws2
					has_tech = mpiws3
				}
			}
		}
		else_if = {
			limit = {
				ROOT = {
					has_tech = mpiws3
					NOT = {
						has_tech = mpiws4
					}
				}
			}
			FROM = {
				OR = {
					has_tech = mpiws3
					has_tech = mpiws4
				}
			}
		}
		#no more steps needed because end of research tree is reached
	}
}
